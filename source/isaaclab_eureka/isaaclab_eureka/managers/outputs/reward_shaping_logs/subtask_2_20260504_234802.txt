        distance_reward = max(last_distance - current_distance, 0) * 1.0  # Increased reward scaling
        if last_distance > 1.0 and current_distance <= 1.0:
            distance_reward += 1.0  # Bonus for entering 1m range
        if last_distance > 0.5 and current_distance <= 0.5:
            distance_reward += 1.0  # Bonus for entering 0.5m range
        if last_distance > 0.2 and current_distance <= 0.2:
            distance_reward += 1.0  # Bonus for entering 0.2m range
        rewards_dict["distance_reward"] = distance_reward

        # Holding Reward: One-time reward for picking up the target object
        if env.last_obs and env.last_obs["is_holding_target"][0] == 1.0:
            holding_reward = 10.0  # Increased moderate reward for picking up the object
            rewards_dict["holding_reward"] = holding_reward

        # Visibility Reward: Reward increase in visibility

        rewards_dict["visibility_reward"] = visibility_reward

        # CounterTop Placement Reward: Reward for placing the mug on a countertop
        interacted_objects = env.get_interacted_objects()
