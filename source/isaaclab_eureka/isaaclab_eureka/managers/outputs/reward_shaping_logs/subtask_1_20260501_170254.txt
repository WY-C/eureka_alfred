    metadata = env.controller.last_event.metadata
    target_object = env.find_target_object()

    rewards_dict = {}

    # Step Penalty
    step_penalty = -0.01

    # Distance Reward
    if target_object:
        distance_to_target = target_object["distance"]

    else:
        distance_reward = 0.0
        rewards_dict["distance_reward"] = distance_reward

    # Visibility Reward

    rewards_dict["visibility_reward"] = visibility_reward

    rewards_dict["pick_up_reward"] = pick_up_reward

    # Total Reward
