    # Step Penalty
    step_penalty = -0.01  # Small negative reward per step

    # Total Reward
    total_reward = proximity_reward + visibility_reward + pickup_reward + step_penalty

    # Reward Components Dictionary

    }

    return total_reward, rewards_dict
