import torch

def _get_rewards_eureka(env):
    metadata = env.controller.last_event.metadata
    total_reward = 0
    rewards_dict = {}

    # Reward for being close to the mug
    mug_objects = [obj for obj in metadata["objects"] if "Mug" in obj["objectType"]]
    closest_mug_distance = float('inf')
    for mug in mug_objects:
        if mug["visible"]:
            closest_mug_distance = min(closest_mug_distance, mug["distance"])

    if closest_mug_distance < float('inf'):
        distance_reward = 1 / (closest_mug_distance + 1)  # Ensure no division by zero
        total_reward += distance_reward
        rewards_dict["distance_reward"] = distance_reward

    # Reward for picking up the mug
    picked_up_mug = any(obj["objectType"] == "Mug" for obj in metadata["inventoryObjects"])
    if picked_up_mug:
        pick_up_reward = 10
        total_reward += pick_up_reward
        rewards_dict["pick_up_reward"] = pick_up_reward

    return total_reward, rewards_dict
