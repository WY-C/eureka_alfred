import torch

def _get_rewards_eureka(env):
    metadata = env.controller.last_event.metadata
    object_info = next((obj for obj in metadata["objects"] if obj["objectType"] == "Mug"), None)
    
    if object_info is None:
        return torch.tensor(0.0), {}
    
    # Get positions
    agent_pos = torch.tensor(metadata["agent"]["position"][:2])
    mug_pos = torch.tensor(object_info["position"][:2])
    
    # Distance to Mug
    distance_to_mug = torch.norm(agent_pos - mug_pos)
    proximity_reward = torch.exp(-10 * distance_to_mug)  # Reward decreases exponentially with distance
    
    # Picked Up Mug
    picked_up_mug = any(obj["objectType"] == "Mug" for obj in metadata["inventoryObjects"])
    pick_up_reward = torch.tensor(10.0 if picked_up_mug else 0.0)  # Large reward for picking up the mug
    
    # Movement Penalty
    action = metadata["lastAction"]
    movement_penalty = torch.tensor(-0.1 if action["action"] in ["MoveLeft", "MoveRight", "MoveForward", "MoveBackward"] else 0.0)
    
    # Total Reward
    total_reward = proximity_reward + pick_up_reward + movement_penalty
    
    # Rewards Dictionary
    rewards_dict = {
        "proximity_reward": proximity_reward,
        "pick_up_reward": pick_up_reward,
        "movement_penalty": movement_penalty
    }
    
    return total_reward, rewards_dict
