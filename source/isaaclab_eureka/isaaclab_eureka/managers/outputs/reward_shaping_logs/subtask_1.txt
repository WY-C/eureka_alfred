def _get_rewards_eureka(env_wrapper):
    metadata = env_wrapper.controller.last_event.metadata

    # Find the target object
    target_object = env_wrapper.find_target_object()
    
    rewards_dict = {}

    # Precondition: Check if the target object is visible
    is_visible = 0
    if target_object is not None:
        is_visible = target_object["visible"]

    # Reward for object visibility
    visibility_reward = is_visible * 2.0  # Increased scale to prioritize visibility

    # Reward for distance to object (scaled down)
    distance_to_object = 0.0
    if target_object is not None:
        distance_to_object = max(0.1, target_object["distance"])  # Avoid division by zero
    distance_reward = 1.0 / distance_to_object  # Reward inversely proportional to distance

    # Reward for picking up the object
    picked_up = 0
    inventory_objects = metadata.get('inventoryObjects', [])
    for obj in inventory_objects:
        if obj['objectType'] == env_wrapper.target_object_type:
            picked_up = 1
            break

    pick_up_object_reward = picked_up * 5.0  # Increased reward for picking up the object

    # Reward for placing object on countertop
    placed_on_counter = 0
    interacted_objects = env_wrapper.get_interacted_objects()
    for obj in interacted_objects["opened"] + interacted_objects["toggled"] + interacted_objects["broken"]:
        if obj.get("parentReceptacles") and any("CounterTop" in p for p in obj.get("parentReceptacles")):
            placed_on_counter = 1
            break

    place_object_on_receptacle_reward = placed_on_counter * 10.0  # High reward for placing on countertop

    # Total reward
    total_reward = (
        visibility_reward +
        distance_reward +
        pick_up_object_reward +
        place_object_on_receptacle_reward
    )

    # Store the reward components in the dictionary
    rewards_dict = {
        "visibility_reward": visibility_reward,
        "distance_reward": distance_reward,
        "pick_up_object_reward": pick_up_object_reward,
        "place_object_on_receptacle_reward": place_object_on_receptacle_reward
    }

    return total_reward, rewards_dict
