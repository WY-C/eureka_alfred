import numpy as np

def _get_rewards_eureka(env):
    metadata = env.controller.last_event.metadata
    target = env.find_target_object()

    rewards_dict = {}

    # Distance Reward
    if target and target["visible"]:
        distance = target["distance"]
        distance_reward = np.exp(-0.5 * distance)
    else:
        distance_reward = 0.0

    rewards_dict["distance_reward"] = distance_reward

    # Visibility Reward
    visible = target["visible"] if target else 0
    visibility_reward = 1.0 if visible else 0.0

    rewards_dict["visibility_reward"] = visibility_reward

    # Pickup Reward
    inventory_objects = metadata.get("inventoryObjects", [])
    picked_up_target = any(obj["objectType"] == env.target_object_type for obj in inventory_objects)
    pickup_reward = 10.0 if picked_up_target else 0.0

    rewards_dict["pickup_reward"] = pickup_reward

    total_reward = distance_reward + visibility_reward + pickup_reward

    return total_reward, rewards_dict
