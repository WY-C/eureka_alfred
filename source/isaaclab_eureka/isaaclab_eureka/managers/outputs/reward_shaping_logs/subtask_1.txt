import numpy as np

def _get_rewards_eureka(env):
    metadata = env.controller.last_event.metadata
    target_object_type = env.target_object_type
    
    rewards_dict = {}
    
    # Find the target object (mug) in the metadata
    target_obj = next((obj for obj in metadata["objects"] if obj["objectType"] == target_object_type), None)
    
    # Check if the object is in the inventory (picked up)
    picked_up = any(obj["objectType"] == target_object_type for obj in metadata["inventoryObjects"])
    pick_up_object_reward = 10.0 if picked_up else 0.0
    
    # Progress reward based on distance to the target object
    progress_reward = 0.0
    if target_obj and target_obj["visible"] == 1:
        distance_to_target = target_obj["distance"]
        progress_reward = np.exp(-distance_to_target)  # Closer -> higher reward
    
    # Penalty for invalid actions
    invalid_action_penalty = -1.0 if env.controller.last_event.metadata["lastActionSuccess"] == False else 0.0
    
    # Total reward calculation
    total_reward = pick_up_object_reward + progress_reward + invalid_action_penalty
    
    # Populate the rewards dictionary
    rewards_dict["pick_up_object_reward"] = pick_up_object_reward
    rewards_dict["progress_reward"] = progress_reward
    rewards_dict["invalid_action_penalty"] = invalid_action_penalty
    
    return total_reward, rewards_dict
