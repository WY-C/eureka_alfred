import torch

def _get_rewards_place_mug(metadata: dict) -> Tuple[torch.Tensor, Dict[str, torch.Tensor]]:
    # Extract relevant information from metadata
    objects = metadata.get("objects", [])
    inventory_objects = metadata.get("inventoryObjects", [])

    # Check if the mug is in the agent's hand
    mug_in_hand = any(obj["objectType"].startswith("Mug") for obj in inventory_objects)

    # Find the mug and the countertop positions
    mug_pos = None
    counter_pos = None
    for obj in objects:
        if obj["objectType"].startswith("Mug"):
            mug_pos = torch.tensor(obj["position"])
        elif obj["objectType"] == "CounterTop":
            counter_pos = torch.tensor(obj["position"])

    # Calculate distance between mug and countertop
    if mug_pos is not None and counter_pos is not None:
        distance_to_counter = torch.norm(mug_pos - counter_pos)
    else:
        distance_to_counter = float('inf')

    # Check if the mug is on the countertop
    mug_on_counter = any(
        obj["objectType"].startswith("Mug") and "CounterTop" in obj.get("parentReceptacles", [])
        for obj in objects
    )

    # Reward components
    distance_reward = torch.exp(-10 * distance_to_counter)  # Increased temperature parameter
    mug_in_hand_reward = float(mug_in_hand)
    mug_on_counter_reward = float(mug_on_counter)

    # Total reward
    total_reward = 5 * distance_reward + 2 * mug_in_hand_reward + 10 * mug_on_counter_reward

    rewards_dict = {
        "distance_reward": distance_reward,
        "mug_in_hand_reward": mug_in_hand_reward,
        "mug_on_counter_reward": mug_on_counter_reward
    }

    return total_reward, rewards_dict
