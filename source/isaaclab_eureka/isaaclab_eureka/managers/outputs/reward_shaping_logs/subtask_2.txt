    # 1. Reward for picking up the target object
    picked_up_reward = 0.0
    if any(obj["objectType"] == target_object_type for obj in interacted_objects["inventory"]):
        picked_up_reward = 10.0
    rewards_dict["picked_up_reward"] = picked_up_reward
    total_reward += picked_up_reward

    # 2. Proximity reward when holding the target object
    proximity_reward = 0.0
    if picked_up_reward > 0:
        target_receptacle = None
        for obj in metadata["objects"]:
            if obj["objectType"] == "CounterTop" and obj["visible"]:
                target_receptacle = obj
                break

        if target_receptacle:
            distance_to_receptacle = target_receptacle["distance"]
            proximity_reward = max(0.0, 1.0 - (distance_to_receptacle / 1.0))  # Scale proximity reward between 0 and 1
            proximity_reward *= 5.0  # Scale reward
        else:
            proximity_reward = -0.1  # Penalize if target receptacle is not visible

    rewards_dict["proximity_reward"] = proximity_reward
    total_reward += proximity_reward

    # 3. Reward for placing the target object on the target receptacle
    placed_reward = 0.0
    for obj in interacted_objects["inventory"]:
        if obj["objectType"] == target_object_type:
            if any("CounterTop" in parent for parent in obj["parentReceptacles"]):
                placed_reward = 20.0
                break
    rewards_dict["placed_reward"] = placed_reward
    total_reward += placed_reward

    # 4. Negative reward for moving away from the target object when not holding it
    move_away_penalty = 0.0
    if picked_up_reward == 0.0 and target_object:
        current_distance = target_object["distance"]
        if wrapper.last_obs is not None:
            last_distance = wrapper.last_obs["distance"][0]
            if current_distance > last_distance:
                move_away_penalty = -1.0

    rewards_dict["move_away_penalty"] = move_away_penalty
    total_reward += move_away_penalty

    # 5. Negative reward for not making progress towards the goal
    no_progress_penalty = -0.05
    rewards_dict["no_progress_penalty"] = no_progress_penalty
    total_reward += no_progress_penalty

    # 6. Penalty for taking too many steps
    step_penalty = -0.01
    rewards_dict["step_penalty"] = step_penalty
    total_reward += step_penalty

    return total_reward, rewards_dict
