def _get_rewards_eureka(wrapper):
    metadata = wrapper.controller.last_event.metadata
    target = wrapper.find_target_object()
    
    rewards_dict = {}

    # Distance Reward: Encourage getting closer to the target object
    distance_reward = 0
    if target:
        distance = target["distance"]
        distance_reward = np.exp(-20 * distance)
        rewards_dict["distance_reward"] = distance_reward

    # Visibility Reward: Encourage the target object to be visible
    visibility_reward = 0
    if target and target["visible"]:
        visibility_reward = 1.0
        rewards_dict["visibility_reward"] = visibility_reward

    # Pickup Reward: Encourage picking up the target object
    pickup_reward = 0
    inventory_objects = metadata.get('inventoryObjects', [])
    if any(obj["objectType"] == wrapper.target_object_type for obj in inventory_objects):
        pickup_reward = 1.0
        rewards_dict["pickup_reward"] = pickup_reward

    # Place Reward: Encourage placing the target object on the receptacle
    place_reward = 0
    objects = metadata.get('objects', [])
    if any(obj["objectType"] == wrapper.target_object_type and obj["parentReceptacles"] and any("CounterTop" in p for p in obj["parentReceptacles"]) for obj in objects):
        place_reward = 10.0  # High reward for successful placement
        rewards_dict["place_reward"] = place_reward

    # Total Reward
    total_reward = distance_reward + visibility_reward + pickup_reward + place_reward

    return total_reward, rewards_dict
