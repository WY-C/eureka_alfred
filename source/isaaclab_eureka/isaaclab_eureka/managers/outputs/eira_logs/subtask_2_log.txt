def _get_rewards_eureka(env):
    metadata = env.controller.last_event.metadata
    total_reward = 0
    rewards_dict = {}

    # Check if the apple is in the inventory
    if any(obj["objectType"] == "Apple" for obj in metadata["inventoryObjects"]):
        total_reward += 1
        rewards_dict["apple_in_inventory"] = 1

    # Check if the apple is on a plate
    if any(
        obj["objectType"] == "Apple" and obj["parentReceptacles"] and
        any("Plate" in p for p in obj["parentReceptacles"])
        for obj in metadata["objects"]
    ):
        total_reward += 10
        rewards_dict["apple_on_plate"] = 10

    # Check if the plate is on the countertop
    if any(
        obj["objectType"] == "Plate" and obj["parentReceptacles"] and
        any("CounterTop" in p for p in obj["parentReceptacles"])
        for obj in metadata["objects"]
    ):
        total_reward += 5
        rewards_dict["plate_on_countertop"] = 5

    return total_reward, rewards_dict
