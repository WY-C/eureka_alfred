def _get_rewards_eureka(env):
    total_reward = 0
    rewards_dict = {
        "approach_apple": 0,
        "grasp_apple": 0,
        "lift_apple": 0,
        "approach_plate": 0,
        "place_apple_on_plate": 0,
        "approach_counter": 0,
        "place_plate_on_counter": 0
    }
    
    # Approach the Apple
    if not env.is_grasped("apple"):
        distance_to_apple = env.get_distance_to("apple")

            total_reward += rewards_dict["approach_apple"]
    
    # Grasp the Apple
    if env.is_grasped("apple"):
        rewards_dict["grasp_apple"] = 2
        total_reward += 2
    
    # Lift the Apple
    if env.is_lifted("apple"):
        rewards_dict["lift_apple"] = 3
        total_reward += 3
    
    # Approach the Plate
    if env.is_grasped("apple") and not env.is_placed_on("apple", "plate"):

        if distance_to_plate < 0.5:  # Adjust threshold as needed
            rewards_dict["approach_plate"] = 1 - distance_to_plate / 0.5
            total_reward += rewards_dict["approach_plate"]
    
    # Place the Apple on the Plate
    if env.is_placed_on("apple", "plate"):

        total_reward += 5
